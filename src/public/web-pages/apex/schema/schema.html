Schema

Schema is a special Apex namespace that gives you access to metadata information about objects and fields in your org.

Schema = Structure (blueprint/table ) of Salesforce data.

It Consist of three parts

 ├── SObjectType (Object Token)

        Represents an object reference (Account, Contact, Opportunity).

 ├── SObjectField (Field Token)

        Represents a field reference.

 └── Describe Results

        Tokens only point to metadata(Object and Field Description)

//Object Token

Schema.SObjectType objToken = Account.SObjectType;

SObject acc = objToken.newSObject();

acc.put('Name','Test Account');

insert acc;

//Field Token

Account accVar = [Select Id,Name from Account limit 1];

Schema.SObjectField fieldToken = Account.Name;

System.debug(accVar.get(fieldToken));

//Explicit Static Description

Schema.DescribeSObjectResult accDescribe2 = Account.SObjectType.getDescribe();

System.debug(accDescribe.isAccessible());

//Field

Schema.DescribeFieldResult fDesc2 = Account.Name.getDescribe();

System.debug(fDesc2.isAccessible());

System.debug(fDesc2.getLength());

System.debug(fDesc2.getLabel());

//Implicit Static Description

Schema.DescribeSObjectResult accDescribe = Schema.SobjectType.Account;

System.debug(accDescribe.isAccessible());

//Field

Schema.DescribeFieldResult fDesc = Schema.SObjectType.Account.fields.Name;

System.debug(fDesc.isAccessible());

System.debug(fDesc.getLabel());

System.debug(fDesc.getType());

//Dynamic Description

Map<String, Schema.SObjectType> gd = Schema.getGlobalDescribe();

Schema.SObjectType objType = gd.get('Account');

Schema.DescribeSObjectResult accDescribe3 = objType.getDescribe();

System.debug(accDescribe3.isAccessible());

//Field

Map<String, Schema.SObjectField> fieldsMap = accDescribe3.fields.getMap();

Schema.DescribeFieldResult fDesc3 = fieldsMap.get('Name').getDescribe();

System.debug(fDesc3.isAccessible());

System.debug(fDesc3.getType());
